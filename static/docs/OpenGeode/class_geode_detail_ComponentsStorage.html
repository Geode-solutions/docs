template class geode::detail::ComponentsStorage

#include <components_storage.h>

template <typename Component>
class ComponentsStorage
{
public:
    // typedefs

    typedef std::unique_ptr<Component> ComponentPtr;
    typedef absl::flat_hash_map<uuid, ComponentPtr> ComponentsStore;
    typedef typename ComponentsStore::const_iterator Iterator;

    // methods

    index_t nb_components() const;
    const Component& component(const uuid& id) const;
    Component& component(const uuid& id);
    Iterator begin() const;
    Iterator end() const;
    void add_component(ComponentPtr component);
    void save_components(absl::string_view filename) const;
    void delete_component(const uuid& id);
    void load_components(absl::string_view filename);
    absl::flat_hash_map<std::string, std::string> file_mapping(absl::string_view directory) const;
};
