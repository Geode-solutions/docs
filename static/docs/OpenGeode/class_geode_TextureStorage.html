template class geode::TextureStorage

#include <texture_storage.h>

template <index_t dimension>
class TextureStorage
{
public:
    // construction

    TextureStorage();
    TextureStorage(TextureStorage&&);

    // methods

    index_t nb_textures(TextureManagerKey) const;

    Texture<dimension>& find_or_create_texture(
        AttributeManager& manager,
        absl::string_view name,
        TextureManagerKey
        );

    const Texture<dimension>& find_texture(
        absl::string_view name,
        TextureManagerKey
        ) const;

    absl::FixedArray<absl::string_view> texture_names(TextureManagerKey) const;
    bool texture_exists(absl::string_view name, TextureManagerKey) const;
    void delete_texture(absl::string_view name, TextureManagerKey);
};
