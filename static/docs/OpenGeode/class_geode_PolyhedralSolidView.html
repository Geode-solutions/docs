template class geode::PolyhedralSolidView

A mesh view of a PolyhedralSolid meaning a subset of a PolyhedralSolid without copying any geometry or microtopology, it only “views” the selected subeset and acts like it is a complete mesh with all the API from vertices to polyhedra, attributes included. Modification on the PolyhedralSolid will not be reproduced on

the corresponding view.

#include <polyhedral_solid_view.h>

public:

    virtual absl::string_view native_extension() const;
    index_t viewed_vertex(index_t vertex_id) const;

    index_t add_viewed_vertex(
        index_t vertex_id,

        );

    index_t viewed_polyhedron(index_t polyhedron_id) const;

    index_t add_viewed_polyhedron(
        index_t polyhedron_id,
        PolyhedralSolidViewKey
        );

    static MeshImpl impl_name_static();
};
