#include <generic_surface_accessor.h>

template <index_t dimension>
class GenericMeshAccessor<SurfaceMesh<dimension>>
{
public:
    // typedefs

    typedef PolygonVertex ElementVertex;
    typedef PolygonVertices ElementVertices;
    typedef PolygonEdge ElementFacet;
    typedef std::array<index_t, 2> ElementFacetVertices;

    // construction

    GenericMeshAccessor(const SurfaceMesh<dimension>& mesh);

    // methods

    index_t nb_elements() const;
    index_t nb_element_vertices(index_t polygon_id) const;
    index_t nb_element_facets(index_t polygon_id) const;
    index_t element_vertex(const ElementVertex& polygon_vertex) const;
    ElementVertices element_vertices(index_t polygon_id) const;
    Point<dimension> element_barycenter(index_t polygon_id) const;
    ElementFacetVertices element_facet_vertices(const ElementFacet& polygon_edge) const;
    absl::optional<index_t> element_adjacent(const ElementFacet& polygon_edge) const;
    absl::optional<ElementFacet> element_adjacent_facet(const ElementFacet& polygon_edge) const;
    const uuid& id() const;
    const Point<dimension>& point(index_t vertex_id) const;
    AttributeManager& element_attribute_manager() const;
};
