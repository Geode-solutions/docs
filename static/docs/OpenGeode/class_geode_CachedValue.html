template class geode::CachedValue

#include <cached_value.h>

template <typename ReturnType>
class CachedValue
{
public:
    // typedefs

    typedef typename std::add_pointer<ReturnType(Args...)>::type CachedFunction;

    // methods

    template <typename... Args>
    const ReturnType& operator () (
        CachedFunction<Args...> computer,
        Args&&... args
        ) const;

    bool operator != (const CachedValue& other) const;
    void reset();
    bool computed() const;
    const ReturnType& value() const;
};
