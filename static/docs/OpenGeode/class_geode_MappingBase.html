template class geode::MappingBase

#include <mapping.h>

template <typename T1, typename T2, template<typename> class StorageType>
class MappingBase
{
public:
    // typedefs

    typedef typename StorageType<T>::Type Storage;

    // methods

    void reserve(index_t capacity);
    bool has_mapping_input(const T1& value) const;
    bool has_mapping_output(const T2& value) const;
    const Storage<T2>& in2out(const T1& in) const;
    const Storage<T1>& out2in(const T2& out) const;
};

// direct descendants

template <typename T1, typename T2 = T1>
class BijectiveMapping;

template <typename T1, typename T2 = T1>
class GenericMapping;
