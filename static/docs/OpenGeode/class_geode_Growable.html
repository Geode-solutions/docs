template class geode::Growable

#include <bitsery_archive.h>

template <typename Archive, typename T>
class Growable
{
public:
    // construction

    Growable(std::vector<std::function<void(Archive&, T&)>> serializers);

    Growable(
        std::vector<std::function<void(Archive&, T&)>> serializers,
        std::vector<std::function<void(T&)>> initializers
        );

    // methods

    template <typename Fnc>
    void serialize(
        Archive& ser,
        const T& obj,
        Fnc&& fnc
        ) const;

    template <typename Fnc>
    void deserialize(
        Archive& des,
        T& obj,
        Fnc&& fnc
        ) const;
};
