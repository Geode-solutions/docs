template class geode::VariableAttribute

Read and write interface for variable attribute storage

#include <attribute.h>

template <typename T>
class VariableAttribute: public geode::ReadOnlyAttribute
{
public:
    // methods

    virtual const T& value(index_t element) const;

    void set_value(
        index_t element,
        T value
        );

    template <typename Modifier>
    void modify_value(
        index_t element,
        Modifier&& modifier
        );

    virtual std::shared_ptr<AttributeBase> clone() const;

    virtual void copy(
        const AttributeBase& attribute,
        index_t nb_elements
        );
};
