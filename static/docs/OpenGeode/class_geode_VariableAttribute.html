template class geode::VariableAttribute<bool>

Read and write interface for boolean variable attribute storage. This class removes the custom storage use by the STL for std::vector<bool>.

#include <attribute.h>

template <>
class VariableAttribute<bool>: public geode::ReadOnlyAttribute
{
public:
    // construction

    VariableAttribute(
        bool default_value,
        AttributeProperties properties,
        AttributeBase::AttributeKey
        );

    // methods

    virtual const bool& value(index_t element) const;
    void set_value(index_t element, bool value);
    bool default_value() const;

    template <typename Modifier>
    void modify_value(
        index_t element,
        Modifier&& modifier
        );

    void compute_value(
        index_t from_element,
        index_t to_element,
        AttributeBase::AttributeKey
        );

    void compute_value(
        const AttributeLinearInterpolation& interpolation,
        index_t to_element,
        AttributeBase::AttributeKey
        );
};
