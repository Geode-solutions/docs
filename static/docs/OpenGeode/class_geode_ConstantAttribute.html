template class geode::ConstantAttribute

Read and write interface for constant attribute storage

#include <attribute.h>

template <typename T>
class ConstantAttribute: public geode::ReadOnlyAttribute
{
public:
    // methods

    virtual const T& value(index_t) const;
    const T& value() const;
    void set_value(T value);

    template <typename Modifier>
    void modify_value(Modifier&& modifier);

    virtual std::shared_ptr<AttributeBase> clone() const;

    virtual void copy(
        const AttributeBase& attribute,
        index_t nb_elements
        );
};
