template class geode::ConstantAttribute

Read and write interface for constant attribute storage

#include <attribute.h>

template <typename T>
class ConstantAttribute: public geode::ReadOnlyAttribute
{
public:
    // construction

    ConstantAttribute(
        T value,
        AttributeProperties properties,
        AttributeBase::AttributeKey
        );

    // methods

    virtual const T& value(index_t) const;
    const T& value() const;
    void set_value(T value);
    T default_value() const;

    template <typename Modifier>
    void modify_value(Modifier&& modifier);

    void compute_value(index_t, index_t, AttributeBase::AttributeKey);

    void compute_value(
        const AttributeLinearInterpolation&,
        index_t,
        AttributeBase::AttributeKey
        );
};
