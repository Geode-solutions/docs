template class geode::Triangle

#include <basic_objects.h>

template <index_t dimension>
class Triangle
{
public:
    // construction

    Triangle(
        const Point<dimension>& p0,
        const Point<dimension>& p1,
        const Point<dimension>& p2
        );

    Triangle(const Triangle<dimension>& other);
    Triangle(Triangle<dimension>&& other);

    // methods

    Triangle<dimension>& operator = (const Triangle<dimension>& other);
    Triangle<dimension>& operator = (Triangle<dimension>&& other);
    Point<dimension> barycenter() const;

    template <index_t T = dimension>
    std::enable_if<T==3, Vector3D>::type normal() const;

    template <index_t T = dimension>
    std::enable_if<T==3, Plane>::type plane() const;

    template <index_t T = dimension>
    std::enable_if<T==3, absl::optional<Vector3D>>::type new_normal() const;

    template <index_t T = dimension>
    std::enable_if<T==3, absl::optional<Plane>>::type new_plane() const;

    void set_point(index_t vertex, const Point<dimension>& point);
    const std::array<std::reference_wrapper<const Point<dimension>>, 3>& vertices() const;
};
