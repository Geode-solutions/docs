template class geode::SparseAttribute

Read and write interface for sparse attribute storage. Only store what you use!

#include <attribute.h>

class SparseAttribute: public geode::ReadOnlyAttribute
{
public:

        );

    T default_value() const;

        index_t element,

        );

        index_t to_element,
        AttributeBase::AttributeKey
        );

    void compute_value(
        const AttributeLinearInterpolation& interpolation,
        index_t to_element,
        AttributeBase::AttributeKey
        );

    virtual std::shared_ptr<AttributeBase> clone() const;

    virtual void copy(
        const AttributeBase& attribute,
        index_t nb_elements
        );
};
