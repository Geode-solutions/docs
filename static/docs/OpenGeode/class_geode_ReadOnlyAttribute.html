template class geode::ReadOnlyAttribute

Read only attribute interface that does not need to know the underlying attribute storage type.

#include <attribute.h>

template <typename T>
class ReadOnlyAttribute: public geode::AttributeBase
{
public:
    // methods

    virtual const T& value(index_t element) const = 0;
    virtual absl::string_view type();
    virtual float generic_value(index_t element) const;
    virtual bool is_genericable() const;
};
