class geode::ImplicitStructuralModel

An Implicit Model is a Structural model where each block has a specific attribute to store the stratigraphic coordinates of its vertices.

#include <implicit_structural_model.h>

class ImplicitStructuralModel: public StructuralModel
{
public:
    // construction

    ImplicitStructuralModel();
    ImplicitStructuralModel(ImplicitStructuralModel&& implicit_model);
    ImplicitStructuralModel(StructuralModel&& structural_model);

    // methods

    absl::string_view native_extension() const;

    const Point3D& stratigraphic_coordinates(
        const uuid& component_id,
        index_t vertex_id
        ) const;

    void instantiate_stratigraphic_attribute_on_blocks(ImplicitStructuralModelBuilderKey);

    void set_stratigraphic_coordinates(
        const uuid& component_id,
        index_t vertex_id,
        Point3D value,
        ImplicitStructuralModelBuilderKey
        );

    static constexpr absl::string_view native_extension_static();
};
