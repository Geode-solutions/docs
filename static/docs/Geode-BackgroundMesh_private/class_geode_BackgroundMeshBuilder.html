template class geode::BackgroundMeshBuilder

#include <background_mesh_builder.h>

template <index_t dimension>
class BackgroundMeshBuilder
{
public:
    // methods

    void map_elements_to_cell(const MeshElements& elements, index_t cell_id);

    void set_cell_corner_vertex(
        const typename RegularGrid<dimension>::Index& cell,
        index_t vertex_id
        );

    void set_vertex_on_border(index_t vertex_id);
    void unset_vertex_on_border(index_t vertex_id);

    void add_component_vertex_to_background_mesh_vertex(
        const uuid& mesh_id,
        index_t vertex_id,
        index_t bmesh_v_id
        );

    void add_component_edge_to_background_mesh_edge(
        const uuid& mesh_id,
        index_t edge_id,
        index_t bmesh_edge_id
        );

    void add_component_edge_to_background_mesh_vertices(
        const uuid& mesh_id,
        index_t edge_id,
        absl::Span<const index_t> bmesh_vertex_ids
        );

    void update_mesh_elements(const Mapping<index_t>& mapping);
    void update_all_mesh_elements(index_t nb_grid_cells, index_t nb_elements);
    void fill_holes();
    void activate_cell(index_t cell_id);
};
