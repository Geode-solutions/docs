template class geode::GridInfo

#include <grid_info.h>

template <index_t dimension>
class GridInfo
{
public:
    // methods

    const RegularGrid<dimension>& grid() const;
    index_t one_mesh_element(index_t cell_id) const;
    absl::InlinedVector<index_t, 1> cell_id(index_t mesh_element_id) const;
    bool is_cell_active(index_t cell_id) const;
    index_t grid_vertex_to_mesh_vertex(index_t grid_vertex_id) const;

    void set_grid_vertex_to_mesh_vertex(
        index_t grid_vertex_id,
        index_t mesh_vertex_id,
        GridInfoKey
        );

    void map_element_to_cell(index_t element, index_t cell_id, GridInfoKey);

    void map_elements_to_cell(
        const MeshElements& elements,
        index_t cell_id,
        GridInfoKey
        );

    void activate_cell(index_t cell_id, GridInfoKey);
    void update_mesh_elements(const Mapping<index_t>& mapping, GridInfoKey);
    void update_all_mesh_elements(index_t nb_total_mesh_elements, GridInfoKey);
};
