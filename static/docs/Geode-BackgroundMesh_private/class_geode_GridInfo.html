template class geode::GridInfo

#include <grid_info.h>

template <index_t dimension>
class GridInfo
{
public:
    // methods

    const RegularGrid<dimension>& grid() const;
    const MeshElements& mesh_elements(const GridCellIndex<dimension>& cell) const;
    index_t one_mesh_element(const GridCellIndex<dimension>& cell) const;
    absl::InlinedVector<index_t, 1> cell_id(index_t mesh_element_id) const;
    bool is_cell_active(index_t cell_id) const;
    bool is_cell_active(const GridCellIndex<dimension>& cell) const;
    index_t cell_corner_vertex(const GridCellIndex<dimension>& cell) const;
    bool is_on_border(index_t vertex_id) const;

    void set_cell_corner_vertex(
        const GridCellIndex<dimension>& cell,
        index_t vertex_id,
        GridInfoKey
        );

    void set_vertex_as_border(index_t vertex_id, GridInfoKey);
    void unset_vertex_as_border(index_t vertex_id, GridInfoKey);
    void map_element_to_cell(index_t element, index_t cell_id, GridInfoKey);

    void map_elements_to_cell(
        const MeshElements& elements,
        index_t cell_id,
        GridInfoKey
        );

    void activate_cell(index_t cell_id, GridInfoKey);
    void update_mesh_elements(const Mapping<index_t>& mapping, GridInfoKey);
    void update_all_mesh_elements(index_t nb_total_mesh_elements, GridInfoKey);
};
