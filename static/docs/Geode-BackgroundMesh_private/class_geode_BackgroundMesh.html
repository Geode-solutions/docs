template class geode::BackgroundMesh

#include <background_mesh.h>

template <index_t dimension>
class BackgroundMesh
{
public:
    // methods

    const RegularGrid<dimension>& grid() const;
    absl::optional<typename RegularGrid<dimension>::Indices> cell(const Point<dimension>& point) const;
    const MeshElements& mesh_elements(const typename RegularGrid<dimension>::Index& cell) const;
    index_t one_mesh_element(const typename RegularGrid<dimension>::Index& cell) const;
    absl::InlinedVector<index_t, 1> cell_id(index_t element_id) const;
    bool is_cell_active(index_t cell_id) const;
    bool is_cell_active(const typename RegularGrid<dimension>::Index& cell) const;
    index_t cell_corner_vertex(const typename RegularGrid<dimension>::Index& cell) const;

    void set_cell_corner_vertex(
        const typename RegularGrid<dimension>::Index& cell,
        index_t vertex_id,
        BackgroundMeshKey
        );

    bool is_on_border(index_t vertex_id) const;
    void set_vertex_as_border(index_t vertex_id, BackgroundMeshKey);
    void unset_vertex_as_border(index_t vertex_id, BackgroundMeshKey);
    std::vector<ComponentVertex> component_vertices(index_t background_mesh_vertex) const;
    std::vector<ComponentEdge> component_edges(index_t background_mesh_edge) const;
    std::vector<ComponentEdge> incident_component_edges(index_t background_mesh_vertex) const;
    void map_element_to_cell(index_t element, index_t cell_id, BackgroundMeshKey);

    void map_elements_to_cell(
        const MeshElements& elements,
        index_t cell_id,
        BackgroundMeshKey
        );

    void add_component_vertex_to_background_mesh_vertex(
        const uuid& mesh_id,
        index_t vertex_id,
        index_t bmesh_v_id,
        BackgroundMeshKey
        );

    void add_component_edge_to_background_mesh_edge(
        const uuid& mesh_id,
        index_t edge_id,
        index_t bmesh_edge_id,
        BackgroundMeshKey
        );

    void add_component_edge_to_background_mesh_vertices(
        const uuid& mesh_id,
        index_t edge_id,
        absl::Span<const index_t> bmesh_vertex_ids,
        BackgroundMeshKey
        );

    void update_mesh_elements(const Mapping<index_t>& mapping, BackgroundMeshKey);

    void update_all_mesh_elements(
        index_t nb_grid_cells,
        index_t nb_elements,
        BackgroundMeshKey
        );

    void activate_cell(index_t cell_id, BackgroundMeshKey);
};

// direct descendants

class BackgroundMesh2D;
