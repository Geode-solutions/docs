template class geode::DataConstraintManager

#include <data_constraint_manager.h>

template <index_t dimension>
class DataConstraintManager
{
public:
    // construction

    DataConstraintManager();
    DataConstraintManager(DataConstraintManager<dimension>&& other);

    // methods

    void load_value_constraints(absl::string_view filename);

    index_t add_value_constraint(
        Point<dimension> constraint_position,
        double constraint_value,
        double constraint_weight
        );

    void remove_value_constraint(index_t value_data_index);
    void change_value_constraint_value(index_t value_data_index, double new_value);
    void change_value_constraint_weight(index_t value_data_index, double new_weight);
    index_t nb_value_constraints() const;
    double value_constraints_mean_value() const;
    const Point<dimension>& value_constraint_point(index_t value_data_index) const;
    double value_constraint_value(index_t value_data_index) const;
    double value_constraint_weight(index_t value_data_index) const;
    void save_value_constraint_manager(absl::string_view filename) const;
};
