A Cross Section is a Section composed of Faults and Horizons (as Lines) and FaultBlocks and StratigraphicUnits (as Surfaces).

#include <cross_section.h>

class CrossSection:
    public Section,
    public Faults2D,
    public Horizons2D,
    public FaultBlocks2D,
    public StratigraphicUnits2D
{
public:
    // typedefs

    typedef CrossSectionBuilder Builder;

    // classes

    class FaultBlockItemRange;
    class FaultItemRange;
    class HorizonItemRange;
    class StratigraphicUnitItemRange;

    // fields

    static constexpr index_t dim { 2 };

    // construction

    CrossSection();
    CrossSection(CrossSection&&);
    CrossSection(Section&& section);

    // methods

    absl::string_view native_extension() const;
    HorizonItemRange horizon_items(const Horizon2D& horizon) const;
    FaultItemRange fault_items(const Fault2D& fault) const;
    FaultBlockItemRange fault_block_items(const FaultBlock2D& fault_block) const;
    StratigraphicUnitItemRange stratigraphic_unit_items(const StratigraphicUnit2D& stratigraphic_unit) const;
    static constexpr absl::string_view native_extension_static();
};

class ImplicitCrossSection;
