class geode::CrossSection

A Cross Section is a Section composed of Faults and Horizons (as Lines) and FaultBlocks and StratigraphicUnits (as Surfaces).

#include <cross_section.h>

class CrossSection:
    public Section,
    public AddComponents< 2, Faults, Horizons, FaultBlocks, StratigraphicUnits >
{
public:
    // classes

    class FaultBlockItemRange;
    class FaultItemRange;
    class HorizonItemRange;
    class StratigraphicUnitItemRange;

    // methods

    static constexpr absl::string_view native_extension_static();
    absl::string_view native_extension() const;
    HorizonItemRange items(const Horizon2D& horizon) const;
    FaultItemRange items(const Fault2D& fault) const;
    FaultBlockItemRange items(const FaultBlock2D& fault_block) const;
    StratigraphicUnitItemRange items(const StratigraphicUnit2D& stratigraphic_unit) const;
};
